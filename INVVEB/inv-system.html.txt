    <script>
        // DATABÁZE UŽIVATELŮ - KAŽDÝ MÁ SVŮJ OBSAH!
        let users = {
            "AAAdmin6": {
                email: "admin@inv.system",
                password: "Arifaj123",
                role: "admin",
                approved: true
            }
        };

        let currentUser = null;
        let editingUser = null; // Uživatel, kterého právě edituje admin

        // NAČTENÍ Z LOCALSTORAGE
        function loadFromStorage() {
            const saved = localStorage.getItem('invUsers');
            if (saved) {
                const parsed = JSON.parse(saved);
                for (const email in parsed) {
                    users[email] = parsed[email];
                }
            }
        }

        // UKLÁDÁNÍ DO LOCALSTORAGE
        function saveToStorage() {
            localStorage.setItem('invUsers', JSON.stringify(users));
        }

        // INICIALIZACE
        loadFromStorage();

        // PŘEPÍNÁNÍ ZÁLOŽEK
        function showTab(tab) {
            document.getElementById('loginTab').classList.toggle('hidden', tab !== 'login');
            document.getElementById('registerTab').classList.toggle('hidden', tab !== 'register');
            
            document.querySelectorAll('.auth-tab').forEach(btn => {
                btn.classList.toggle('active', btn.textContent.includes(tab === 'login' ? 'Přihlášení' : 'Registrace'));
            });
        }

        // REGISTRACE
        function register() {
            const email = document.getElementById('regEmail').value;
            const password = document.getElementById('regPassword').value;
            const confirm = document.getElementById('regPasswordConfirm').value;
            const error = document.getElementById('regError');

            if (!email || !password) {
                error.textContent = "Vyplňte všechny údaje!";
                return;
            }

            if (password !== confirm) {
                error.textContent = "Hesla se neshodují!";
                return;
            }

            if (users[email]) {
                error.textContent = "Uživatel již existuje!";
                return;
            }

            // VYTVOŘENÍ NOVÉHO UŽIVATELE S JEHO VLASTNÍM OBSAHEM
            users[email] = {
                email: email,
                password: password,
                role: "user",
                approved: false,
                content: {
                    uvod: { text: "Vítejte na úvodní stránce! Tento obsah je přizpůsobený právě pro vás." },
                    zakazky: { text: "Zde uvidíte své zakázky...", image: "" },
                    banka: { text: "Vaše bankovní informace...", image: "" },
                    kontakt: { text: "Kontaktní informace...", image: "" }
                }
            };

            saveToStorage();
            error.textContent = "Registrace úspěšná! Čekejte na schválení administrátorem.";
            document.getElementById('regEmail').value = "";
            document.getElementById('regPassword').value = "";
            document.getElementById('regPasswordConfirm').value = "";
        }

        // PŘIHLÁŠENÍ
        function login() {
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            const error = document.getElementById('loginError');

            if (!users[email]) {
                error.textContent = "Uživatel neexistuje!";
                return;
            }

            if (users[email].password !== password) {
                error.textContent = "Nesprávné heslo!";
                return;
            }

            if (!users[email].approved && users[email].role !== 'admin') {
                error.textContent = "Účet čeká na schválení administrátorem!";
                return;
            }

            currentUser = users[email];
            error.textContent = "";

            // PŘEPNUTÍ NA SPRÁVNOU APLIKACI
            document.getElementById('authPanel').classList.add('hidden');
            
            if (currentUser.role === 'admin') {
                document.getElementById('adminAppPanel').classList.remove('hidden');
                loadUserList();
            } else {
                document.getElementById('userAppPanel').classList.remove('hidden');
                loadCurrentUserContent();
            }
        }

        // NAČTENÍ OBSAHU PRO NORMÁLNÍHO UŽIVATELE
        function loadCurrentUserContent() {
            if (!currentUser || !currentUser.content) return;

            const contentMap = {
                'uvod': 'userUvodContent',
                'zakazky': 'userZakazkyContent', 
                'banka': 'userBankaContent',
                'kontakt': 'userKontaktContent'
            };

            for (const [page, elementId] of Object.entries(contentMap)) {
                const element = document.getElementById(elementId);
                if (element && currentUser.content[page]) {
                    element.innerHTML = currentUser.content[page].text;
                }
            }
        }

        // NAČTENÍ SEZNAMU UŽIVATELŮ (pro admina)
        function loadUserList() {
            const userList = document.getElementById('userList');
            userList.innerHTML = '';

            for (const email in users) {
                if (users[email].role === 'user') {
                    const user = users[email];
                    const userCard = document.createElement('div');
                    userCard.className = `user-card ${user.approved ? '' : 'pending'}`;
                    userCard.innerHTML = `
                        <h4>${email}</h4>
                        <p><strong>Heslo:</strong> ${user.password}</p>
                        <p><strong>Stav:</strong> ${user.approved ? 'Schválen' : 'Čeká na schválení'}</p>
                        <div class="user-actions">
                            ${!user.approved ? `<button onclick="approveUser('${email}')">Schválit</button>` : ''}
                            <button class="edit-user-btn" onclick="editUserContent('${email}')">Upravit obsah</button>
                            <button onclick="deleteUser('${email}')" style="background: #e74c3c;">Smazat</button>
                        </div>
                    `;
                    userList.appendChild(userCard);
                }
            }
        }

        // EDITACE OBSAHU UŽIVATELE (admin funkce)
        function editUserContent(email) {
            editingUser = users[email];
            document.getElementById('editingUserEmail').textContent = email;
            document.getElementById('userEditor').classList.remove('hidden');

            // NAČTENÍ OBSAHU UŽIVATELE DO EDITORU
            loadUserContentToEditor();
        }

        // NAČTENÍ OBSAHU DO EDITORU
        function loadUserContentToEditor() {
            if (!editingUser || !editingUser.content) return;

            const contentMap = {
                'uvod': 'adminUvodContent',
                'zakazky': 'adminZakazkyContent',
                'banka': 'adminBankaContent', 
                'kontakt': 'adminKontaktContent'
            };

            for (const [page, elementId] of Object.entries(contentMap)) {
                const element = document.getElementById(elementId);
                if (element && editingUser.content[page]) {
                    element.innerHTML = editingUser.content[page].text;
                }
            }
        }

        // UKLÁDÁNÍ OBSAHU UŽIVATELE (admin funkce)
        function saveUserContent(page) {
            if (!editingUser) return;

            const contentElement = document.getElementById(`admin${page.charAt(0).toUpperCase() + page.slice(1)}Content`);
            if (contentElement) {
                editingUser.content[page].text = contentElement.innerHTML;
                saveToStorage();
                alert(`Obsah pro stránku ${page} byl uložen pro uživatele ${editingUser.email}!`);
            }
        }

        // PŘEPÍNÁNÍ ZÁLOŽEK V EDITORU
        function showEditorTab(tab) {
            document.querySelectorAll('.editor-content').forEach(content => content.classList.add('hidden'));
            document.querySelectorAll('.editor-tab').forEach(btn => btn.classList.remove('active'));
            
            document.getElementById(`editor${tab.charAt(0).toUpperCase() + tab.slice(1)}Tab`).classList.remove('hidden');
            event.target.classList.add('active');
        }

        // SCHVÁLENÍ UŽIVATELE
        function approveUser(email) {
            users[email].approved = true;
            saveToStorage();
            loadUserList();
            alert(`Uživatel ${email} byl schválen!`);
        }

        // SMAZÁNÍ UŽIVATELE
        function deleteUser(email) {
            if (confirm(`Opravdu chcete smazat uživatele ${email}?`)) {
                delete users[email];
                saveToStorage();
                loadUserList();
            }
        }

        // PŘEPÍNÁNÍ STRÁNEK
        function showPage(page) {
            document.querySelectorAll('.content-page').forEach(p => p.classList.remove('active'));
            document.querySelectorAll('.nav-button').forEach(btn => btn.classList.remove('active'));
            
            document.getElementById(page + 'Page').classList.add('active');
            event.target.classList.add('active');
        }

        // ODHLAŠENÍ
        function logout() {
            currentUser = null;
            editingUser = null;
            document.getElementById('userAppPanel').classList.add('hidden');
            document.getElementById('adminAppPanel').classList.add('hidden');
            document.getElementById('authPanel').classList.remove('hidden');
            document.getElementById('userEditor').classList.add('hidden');
            document.getElementById('loginEmail').value = "";
            document.getElementById('loginPassword').value = "";
        }
    </script>
